import matplotlib.pyplot as plt
plt.figure(figsize=(20, 15))

countries_of_interest = ['Sweden', 'US', 'Germany', 'Russia', 'Italy', 'Spain']

for i in countries_of_interest:
    cases.loc[i].plot()

plt.xlabel('DATE', fontsize = 22)
plt.ylabel('LOG10 ( CASES )', fontsize = 22)
plt.title('NUMBER OF CONFIRMED CASES', fontsize = 22)
plt.legend(countries_of_interest, loc='upper left', fontsize = 22)
plt.show()
